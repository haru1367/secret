/* 가장긴 연속된 합성수 수열의 최소 소인수 찾기 */
# include <stdio.h>
int main() {

	int i;
	int index = 1;
	int ok;
	int prime[10000];
	int N;
	int j;
	int count = 0;
	int k;
	int M;
	int Teststart;

	prime[0] = 2;
	prime[1] = 3;
	int primenumber = 5;
	for (;;) {
		ok = 0;
		for (i = 0; i <= index; i++) {
			if (primenumber % prime[i] != 0) {
				ok++;

			}
			else {
				break;
			}
		}

		if (ok == index + 1) {
			index++;
			prime[index] = primenumber;
			if (index == 9999) {
				break;
			}
		}
		primenumber += 2;
	}

	printf("숫자를 입력하세요:");
	scanf("%d", &N);
	printf("어디부터 테스트 할건가요?:");
	scanf("%d", &Teststart);
	printf("어디까지 테스트 할건가요?:");
	scanf("%d", &M);
	j = 0;
	for (k = Teststart; k <= M;) {
		if (k % prime[j] == 0) {
			count += 1;
			printf("%d ", prime[j]);
			k++;
			j = 0;
		}
		else if (k % prime[j] != 0 && j < N - 1) {
			j++;
		}
		else if (j == N - 1) {
			printf("---->%d, 숫자:%d", count, k);
			if (count >= 2 * prime[N - 2] - 1) {
				break;
				printf("해당하는 수열");
			}
			
			
			printf("\n");
			k++;
			j = 0;
			count = 0;
		}
	}
	return 0;
}
